function sentence(List) {
  let result = List.sort((a, b) => Object.keys(a)[0] - Object.keys(b)[0])
  let string = '';
  for (let i = 0; i < result.length; i++){
    let key = Object.keys(result[i])[0]
    string += result[i][key] + ' '
  }
  return string.trim()
}
